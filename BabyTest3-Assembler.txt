; THIS PROGRAM ADDS TWO NUMBERS TOGETHER AND STORES NEGATIVE SUM
; The program starts at address 1, so we need to pad out address 0 
          VAR 0       ; Declare 32-bit variable to fill space 
START:    LDP NUM01   ; Copy variable to accumulator
          ADD NUM02   ; Subtract variable from accumulator to get sum (negated)
          NEGSTO MYSUM   ; Store accumulator to variable - saves answer
          JRP MYSUM   ; Add content of store to CI
END:      STP         ; Stop processor
; Variable declarations follow
NUM01:    VAR 4    ; Declare 32-bit variable
NUM02:    VAR 12   ; Declare 32-bit variable
MYSUM:    VAR 0    ; Declare 32-bit variable
